I developed a bash script to decode the passwords in base64.

```bash
#!/bin/bash

for x in $(cat mysql.users); do
        db64=$(echo $x | cut -d ':' -f 2 | base64 -d)
        echo -e "$x  (\e[1;33m$db64\e[0m)"
done
```
